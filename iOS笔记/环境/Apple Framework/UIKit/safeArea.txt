// 顶部安全区高度
/**
window:(顶部状态栏)
iOS 11系统有点特殊[11.3, 11.4 已测试]，iPhoneX: 44, 但iPhone6s: 0
iOS 12系统，iPhoneX: 44, 但iPhone6s: 20

view:(有navigationbar，顶部状态栏 + 导航栏) iPhone6s: 64
*/

==========iOS 26
只有navBar:
viewDidAppear

statusBarHeight = 54.0
navigationBarHeight = 54.0
view.topSafeAreaHeight = 116.0 // vc.view.safeAreaInsets.top
tabBarHeight = 0.0
view.bottomSafeAreaHeight = 34.0 // vc.view.safeAreaInsets.bottom
window.topSafeAreaHeight = 62.0 // window.safeAreaInsets.top
window.bottomSafeAreaHeight = 34.0 // window.safeAreaInsets.bottom

(statusBarHeight + navigationBarHeight = 108) != (view.topSafeAreaHeight = 116.0)
需要使用
view.topSafeAreaHeight
view.bottomSafeAreaHeight
window.topSafeAreaHeight
window.bottomSafeAreaHeight
navigationBarHeight
tabBarHeight
这些进行布局，顶部（top）布局不能用 statusBarHeight + navigationBarHeight 的高度
使用 view.topSafeAreaHeight = 116
或者 window.topSafeAreaHeight + navigationBarHeight = 62 + 54 = 116
